	DEFAULT CHARACTER SET utf8

USE auction;

CREATE TABLE users(
	user_id INT UNSIGNED AUTO_INCREMENT NOT NULL,
	email VARCHAR(50) NOT NULL UNIQUE,
	first_name VARCHAR(30) NOT NULL,
	last_name VARCHAR(30) NOT NULL,
	password VARCHAR(128) NOT NULL,
	reg_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY(user_id)
);


CREATE TABLE auctions(
	 auction_id
	 user_id
	 current_price
	 item_id
	 starting_price
	 reserve_price
	 status
	 watch_id
	 category_id

);


CREATE TABLE bids(
	bid_id
	auction_id
	user_id
);

CREATE TABLE items(
	item_id
	item_desc
	category_id
	user_id
	condition
	picture
);


CREATE TABLE categories(
	category_id
	category_name
	category_desc
	item_id
	user_id
);

CREATE TABLE reviews(
	review_id
	item_id
	rating
	user_id
	comment
);

CREATE TABLE watch(
	watch_id
	user_id
	auction_id	
);
